   baseGST = baseFare * (classType === "BUSINESS" ? 0.12 : 0.05)

     server/src/utils/calculateFare.js

   Sample code to include:
   ---------------------------------------------------
   module.exports = function calculateFare(baseFare, classType, platformFee = 40) {
     const isBusiness = classType === "BUSINESS";
     const baseGST = baseFare * (isBusiness ? 0.12 : 0.05);
     const serviceGST = platformFee * 0.18;

     return {
       baseFare,
       classType,
       platformFee,
       gst: {
         baseGST: Math.round(baseGST),
         serviceGST: Math.round(serviceGST),
         totalGST: Math.round(baseGST + serviceGST)
       },
       totalAmount: Math.round(baseFare + platformFee + baseGST + serviceGST)
     };
   };
   ---------------------------------------------------

   Ensure the booking POST endpoint uses this function before saving the booking.
